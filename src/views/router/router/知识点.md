# 路由

- 普通跳转

```
<!-- 跳转 -->
<router-link to="/router1">router1</router-link>
<!-- 样式 -->
<router-link to="/router2" tag="div">router2</router-link>
```

- 懒加载

```
{
  path: '/about',
  name: 'About',
  // route level code-splitting
  // this generates a separate chunk (about.[hash].js) for this route
  // which is lazy-loaded when the route is visited.
  // 路由的懒加载，打包会生成一个单独的文件
  // /* webpackChunkName: "about" */是命名,没有名字会自动生成
  component: () => import(/* webpackChunkName: "about" */ '@/views/About.vue')
}
```

- 动态路由

```
<!-- 动态跳转 -->
<router-link to="/router/3">router3</router-link> |
<!-- 动态跳转,生命周期不更新，使用监听watch或检测key之 -->
<router-link to="/router/3-1">router3-1</router-link> |
<router-link to="/router/3-1">router3-1</router-link> |

1、
watch: {
  '$route' (to, from) {
    console.log(to, from);
  }
}
2、
<router-view :key="key" />

computed: {
  key () {
    // 重新生成fullPath，更新生命周期
    return this.$route.fullPath
  }
},
```

- 通配符匹配

```
// 通配符匹配 *
{
  path: "*",
  name: "router3",
  component: () => import("@/views/router/router/router4"),
},
```
